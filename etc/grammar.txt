program        := statements $ ;
statements     := statements statement | ~ ;
statement      := 'i' ":=" expr | "read" 'i' | "write" expr ;
expr           := term | expr add_op term ;
term           := factor | term mult_op factor ;
factor         := ( expr ) | 'i' | 'd' ;
add_op         := + | - ;
mult_op        := * | / ; 

// $ --> end of text
// ~ --> empty, nothing
