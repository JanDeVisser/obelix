%
name: simple_expr
init: dummy_init 
ignore_ws: true
%
program        := statements ;
statements     := statement statements | ;
statement      := 'i' ":=" expr | "read" 'i' | "write" expr ;
expr [ init: start_expr ] := term termtail ;
termtail       := add_op term termtail | ;
term           := factor factortail ; 
factortail     := mult_op factor factortail | ;
factor         := ( expr ) | 'i' | 'd' ;
add_op         := + | - ;
mult_op        := * | / ;
